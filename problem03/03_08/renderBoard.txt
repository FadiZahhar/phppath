function renderBoard($layout) {
    echo '<table>';
    for ($row = 0; $row < BOARD_SIZE; $row++) {
        echo '<tr>';
        for ($col = 0; $col < BOARD_SIZE; $col++){
            if ($layout[$row] == 1 << $col) {
                echo '<td><img src="crown.png" width="28" height="26"></td>';
            } else {
                echo '<td>&nbsp;</td>';
            }
        }
        echo '</tr>';
    }
    echo '</table>';
}
